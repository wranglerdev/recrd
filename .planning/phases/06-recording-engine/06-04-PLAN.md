```csharp
using System.Reflection;
using System.Text;
using System.Text.Json;
using Microsoft.Playwright;
using Recrd.Core.Pipeline;

namespace Recrd.Recording.Inspector;

internal sealed class InspectorServer : IAsyncDisposable
{
    private IBrowser? _inspectorBrowser;
    private IBrowserContext? _inspectorContext;
    private IPage? _inspectorPage;
    private bool _isOpen;
    private readonly Func<string, Task> _onInspectorCallback;

    public InspectorServer(Func<string, Task> onInspectorCallback)
    {
        _onInspectorCallback = onInspectorCallback;
    }

    public bool IsOpen => _isOpen;

    public async Task OpenAsync(IPlaywright playwright)
    {
        // Launch secondary browser in app mode (per D-06, REC-11)
        _inspectorBrowser = await playwright.Chromium.LaunchAsync(new()
        {
            Headless = false,
            Args = new[] { "--app=http://recrd-inspector.local/" }
        });

        _inspectorContext = await _inspectorBrowser.NewContextAsync(new()
        {
            ViewportSize = new() { Width = 420, Height = 700 }
        });

        _inspectorPage = await _inspectorContext.NewPageAsync();

        // Register inspector callback BEFORE routing (same order principle as recording context)
        await _inspectorContext.ExposeFunctionAsync("__recrdInspectorCallback", async (string payload) =>
        {
            await _onInspectorCallback(payload);
        });

        // Serve inspector HTML via RouteAsync (per RESEARCH.md Pattern 3)
        var html = LoadEmbeddedHtml();
        await _inspectorPage.RouteAsync("**/*", async route =>
        {
            await route.FulfillAsync(new RouteFulfillOptions
            {
                Status = 200,
                ContentType = "text/html; charset=utf-8",
                Body = html
            });
        });

        await _inspectorPage.GotoAsync("http://recrd-inspector.local/");
        _isOpen = true;

        // Handle inspector being closed by the user
        _inspectorPage.Close += (_, _) => _isOpen = false;
    }

    public async Task PushEventAsync(RecordedEvent evt)
    {
        if (!_isOpen || _inspectorPage is null) return;
        try
        {
            var json = JsonSerializer.Serialize(new
            {
                type = evt.EventType.ToString(),
                selector = evt.Selectors.Count > 0 ? evt.Selectors[0].Values.Values.FirstOrDefault() : "",
                timestamp = $"+{evt.TimestampMs}ms"
            });
            await _inspectorPage.EvaluateAsync($"window.__recrdPush({json})");
        }
        catch (PlaywrightException)
        {
            // Inspector page closed — per RESEARCH.md Pitfall 5, swallow and continue
            _isOpen = false;
        }
    }

    public async Task SetStateAsync(string state)
    {
        if (!_isOpen || _inspectorPage is null) return;
        try
        {
            await _inspectorPage.EvaluateAsync($"window.__recrdSetState('{state}')");
        }
        catch (PlaywrightException) { _isOpen = false; }
    }

    public async Task ShowTagDialogAsync(string selectorJson)
    {
        if (!_isOpen || _inspectorPage is null) return;
        try
        {
            await _inspectorPage.EvaluateAsync($"window.__recrdShowTagDialog({selectorJson})");
        }
        catch (PlaywrightException) { _isOpen = false; }
    }

    public async Task ShowTagErrorAsync(string errorType)
    {
        if (!_isOpen || _inspectorPage is null) return;
        try
        {
            await _inspectorPage.EvaluateAsync($"window.__recrdTagError('{errorType}')");
        }
        catch (PlaywrightException) { _isOpen = false; }
    }

    public async Task ShowTagSuccessAsync(string name)
    {
        if (!_isOpen || _inspectorPage is null) return;
        try
        {
            await _inspectorPage.EvaluateAsync($"window.__recrdTagSuccess('{name}')");
        }
        catch (PlaywrightException) { _isOpen = false; }
    }

    public async Task ShowAssertDialogAsync(string dataJson)
    {
        if (!_isOpen || _inspectorPage is null) return;
        try
        {
            await _inspectorPage.EvaluateAsync($"window.__recrdShowAssertDialog({dataJson})");
        }
        catch (PlaywrightException) { _isOpen = false; }
    }

    private static string LoadEmbeddedHtml()
    {
        var assembly = Assembly.GetExecutingAssembly();
        using var stream = assembly.GetManifestResourceStream("Recrd.Recording.Panel.inspector.html")
            ?? throw new InvalidOperationException("inspector.html not found as embedded resource");
        using var reader = new StreamReader(stream, Encoding.UTF8);
        return reader.ReadToEnd();
    }

    public async ValueTask DisposeAsync()
    {
        _isOpen = false;
        if (_inspectorPage is not null) await _inspectorPage.CloseAsync();
        if (_inspectorContext is not null) await _inspectorContext.CloseAsync();
        if (_inspectorBrowser is not null) await _inspectorBrowser.CloseAsync();
    }
}
```

Add `HandleInspectorCallback` method:
```csharp
private async Task HandleInspectorCallback(string payload)
{
    using var doc = JsonDocument.Parse(payload);
    var root = doc.RootElement;
    var type = root.GetProperty("type").GetString();

    switch (type)
    {
        case "TagConfirm":
            var name = root.GetProperty("name").GetString()!;
            // Validate name against regex
            if (!System.Text.RegularExpressions.Regex.IsMatch(name, @"^[a-z][a-z0-9_]{0,63}$"))
            {
                if (_inspector?.IsOpen == true)
                    await _inspector.ShowTagErrorAsync("invalid");
                return;
            }
            // Check duplicate
            if (_sessionBuilder!.HasVariable(name))
            {
                if (_inspector?.IsOpen == true)
                    await _inspector.ShowTagErrorAsync("duplicate");
                return;
            }
            // Add variable to session
            var variable = new Variable(name);
            _sessionBuilder.AddVariable(variable);
            if (_inspector?.IsOpen == true)
                await _inspector.ShowTagSuccessAsync(name);
            break;

        case "AssertConfirm":
            var assertionTypeStr = root.GetProperty("assertionType").GetString()!;
            var expected = root.GetProperty("expected").GetString() ?? "";
            var assertionType = Enum.Parse<AssertionType>(assertionTypeStr);
            // Build selector from the stored assert-start data
            var selectorJson = root.GetProperty("selector");
            var selector = SelectorExtractor.Extract(selectorJson);
            var assertStep = new AssertionStep(assertionType, selector, new Dictionary<string, string> { ["expected"] = expected }.AsReadOnly());
            _sessionBuilder!.AddStep(assertStep);
            break;
    }
}
```
